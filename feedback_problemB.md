### 3. New and Updated Guidelines that worked for Problem B

#### Guideline #7: Combine Counterexample-First Prompting with a Human-in-the-Loop to Guide the LLM Towards Tests That Pass  
**Guideline Definition**  

Instruct the LLM to write tests for the code and fix the bugs that it finds. First ask the model to propose plausible wrong variants and minimal counterexamples. Then generate tests that pass for the intended contract and would fail for each wrong variant. Next, run the application in the browser. If issues persist, tell the LLM to fix the issue by first writing a test to verify the issue exists, and then writing code to make the test pass. This process ensures that you not only have working code, but you have tests to back it up. Anytime new code is added to the codebase, the existing tests will run to make sure that previous bugs fixed in the past are still working correctly with the newly added code.

**Prompt and Context:**  
Context:  
The code in the ProblemB folder.

Step 1:  
Prompt:  
You are helping me write discriminative node assert unit tests.  
Target: snake.js  
Task: list 4-6 plausible wrong variants that are easy to accidentally implement. For each wrong variant, provide the smallest counterexample inputs that would expose it.  
Return a table: wrong_variant | symptom | counterexample_inputs

Step 2:  
Prompt:  
Using the table above, write node assert unit tests such that:  
1) they pass for the intended contract, and  
2) each test would fail for its corresponding wrong_variant.

Update the code to make any failing tests pass. First write and run the test to verify the issue exists, and then write code to make the test pass.

```
import assert from 'assert';
import { createGame } from './snake.js';


function test(name, fn) {
  try {
    fn();
    console.log('PASS:', name);
  } catch (e) {
    console.error('FAIL:', name, e.message);
  }
}

function setupSnake(snakeArr, food, cols=10, rows=10) {
  const game = createGame(cols, rows);
  game.state.snake = snakeArr.map(([x, y]) => ({x, y}));
  if (food) game.state.food = {x: food[0], y: food[1]};
  game.state.gameState = 'running';
  return game;
}

test('should move snake by one cell (no off-by-one)', () => {
  const game = setupSnake([[5,5]]);
  game.state.nextDirection = {x: 1, y: 0};
  game.update();
  assert.deepStrictEqual([game.state.snake[0].x, game.state.snake[0].y], [6,5]);
});

test('should detect self-collision', () => {
  // Snake forms a loop: head at (5,5), body at (5,6), (6,6), (6,5)
  // Move left, head moves to (4,5), no collision
  // Move down, head moves to (4,6), no collision
  // Move right, head moves to (5,6), collides with body
  const game = setupSnake([[5,5],[5,6],[6,6],[6,5]]);
  game.state.direction = {x: -1, y: 0};
  game.state.nextDirection = {x: -1, y: 0}; // move left
  game.update(); // head at (4,5)
  game.state.direction = {x: 0, y: 1};
  game.state.nextDirection = {x: 0, y: 1}; // move down
  game.update(); // head at (4,6)
  game.state.direction = {x: 1, y: 0};
  game.state.nextDirection = {x: 1, y: 0}; // move right
  game.update(); // head at (5,6), should collide
  assert.strictEqual(game.state.gameState, 'dead');
});

test('should grow when eating food', () => {
  const game = setupSnake([[5,5]], [6,5]);
  game.state.nextDirection = {x: 1, y: 0};
  game.update();
  assert.strictEqual(game.state.snake.length, 2);
});

test('should not grow when not eating', () => {
  const game = setupSnake([[5,5]], [7,5]);
  game.state.nextDirection = {x: 1, y: 0};
  game.update();
  assert.strictEqual(game.state.snake.length, 1);
});

test('should detect wall collision', () => {
  const game = setupSnake([[0,5]]);
  game.state.nextDirection = {x: -1, y: 0};
  game.update();
  assert.strictEqual(game.state.gameState, 'dead');
});

test('should not spawn food on snake body', () => {
  const game = setupSnake([[5,5],[6,5]]);
  game.placeFood();
  const food = [game.state.food.x, game.state.food.y];
  assert.notDeepStrictEqual(food, [5,5]);
  assert.notDeepStrictEqual(food, [6,5]);
});
```

It updates the snake.js code to the following:
```
export const KEY_MAP = {
  ArrowUp: [0, -1], ArrowDown: [0, 1], ArrowLeft: [-1, 0], ArrowRight: [1, 0],
  w: [0, 1], s: [0, -1], a: [-1, 0], d: [1, 0],
  W: [0, 1], S: [0, -1], A: [-1, 0], D: [1, 0],
};

export function createGame(cols, rows, { baseInterval = 120, speedIncrease = 2, minInterval = 55 } = {}) {
  const state = {
    snake: [],
    direction: { x: 0, y: 0 },
    nextDirection: { x: 0, y: 0 },
    food: null,
    score: 0,
    highScore: 0,
    gameState: 'idle',
    interval: baseInterval,
  };

  function init() {
    const midX = Math.floor(cols / 2);
    const midY = Math.floor(rows / 2);
    state.snake = [
      { x: midX, y: midY },
      { x: midX - 1, y: midY },
      { x: midX - 2, y: midY },
    ];
    state.direction = { x: 1, y: 0 };
    state.nextDirection = { x: 1, y: 0 };
    state.score = 0;
    state.interval = baseInterval;
    placeFood();
  }

  function placeFood() {
    const occupied = new Set(state.snake.map(s => `${s.x},${s.y}`));
    let pos;
    do {
      pos = { x: Math.floor(Math.random() * cols), y: Math.floor(Math.random() * rows) };
    } while (occupied.has(`${pos.x},${pos.y}`));
    state.food = pos;
  }

  function update() {
    state.direction = { ...state.nextDirection };
    const head = {
      x: state.snake[0].x + state.direction.x,
      y: state.snake[0].y + state.direction.y,
    };

    if (head.x < 0 || head.x >= cols || head.y < 0 || head.y >= rows) {
      die();
      return;
    }
    // Check collision with body (excluding head)
    if (state.snake.some((s, i) => i > 0 && s.x === head.x && s.y === head.y)) {
      die();
      return;
    }

    state.snake.unshift(head);

    // Self-collision: check for duplicate coordinates
    const headStr = `${head.x},${head.y}`;
    const bodyStrs = state.snake.slice(1).map(s => `${s.x},${s.y}`);
    if (bodyStrs.includes(headStr)) {
      die();
      return;
    }

    if (state.food && head.x === state.food.x && head.y === state.food.y) {
      state.score++;
      state.interval = Math.max(minInterval, baseInterval - state.score * speedIncrease);
      placeFood();
      // Do not pop tail, snake grows
    } else {
      // Only pop tail if not eating food
      state.snake.pop();
    }
  }

  function die() {
    state.gameState = 'dead';
    if (state.score < state.highScore) {
      state.highScore = state.score;
    }
  }

  function setDirection(dx, dy) {
    if (state.direction.x === -dx && state.direction.y === -dy) return;
    if (dx !== 0 && state.nextDirection.x === dx) return;
    if (dy !== 0 && state.nextDirection.y === dy) return;
    state.nextDirection = { x: dx, y: dy };
  }

  function togglePause() {
    if (state.gameState === 'running') {
      state.gameState = 'paused';
    } else if (state.gameState === 'paused') {
      state.gameState = 'running';
    }
  }

  function startGame() {
    init();
    state.gameState = 'running';
  }

  return { state, init, placeFood, update, die, setDirection, togglePause, startGame };
}
```

The application now runs flawlessly in the browser.

**Analysis:**  
In this updated guideline, both the human and the LLM are able to work together to achieve a common goal. That is fixing the existing bugs in the code and leaving behind high quality tests from the bug fixes that can be used to ensure that those bugs remain fixed, even as new code and new features are added to the application.
